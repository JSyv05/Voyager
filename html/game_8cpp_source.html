47 const auto& input = *command.getInput();
48 if (input.empty()) {
49 return MainMenuCommand::Error;
50 }
51 else if ((input.size() == 1 && input[0] == "start") ||
52 (input.size() >= 2 && input[0] == "start" && input[1] == "game")) {
53 return MainMenuCommand::Start;
54 }
55 else if ((input.size() == 1 && input[0] == "load") ||
56 (input.size() >= 2 && input[0] == "load" && input[1] == "game")) {
57 return MainMenuCommand::Load;
